Tools for Numerical Analysis (83214)

The Following code is an example of simple implementation of the 4 algorithms for zero-point finding.

import math
# just for showing in class
import time

# consts #
delay = 0
# end - consts #

def f(x: float) -> float:
""" :param x: the input of the function :return: x^3 + 1 """
return x**3 + 1

def df(x: float) -> float:
""" :param x: the input of the function :return: 3x^2 """
return 3 * x**2

def f1(x: float):
""" :param x: the input of the function :return: sin(x) * x - pi """
return math.sin(x) * x - math.pi

def df1(x: float):
""" :param x: the input of the function :return: cos(x) * x + sin(x) """
return math.cos(x) * x + math.sin(x)

def f2(x: float):
""" :param x: the input of the function :return: e^(-1/x^2) """
return math.exp(-1/(x**2))

def df2(x: float):
""" :param x: the input of the function :return: 3x^2 """
return 2 * math.exp(-1/(x**2)) / x**3

class Bisection:
""" bisection method for a given function """

@staticmethod
def run(function, a: float, b: float, precision: float) -> float:
""" run the bisection method of a function and start conditions [a,b] :param function: the function to run :param a: the smaller location :param b: the upper location :param precision: how accurate we want the answer :return: where function = 0 inside [a,b] """
# first, make sure b > a
if b < a:
temp = a
a = b
b = temp
# make sure the function decrease and then increase
if not (function(b) > 0 > function(a) or function(a) > 0 > function(b)): return "error"
# make the calculation itself
index = 1
while (b-a) > precision:
c = (a + b) / 2
print("#{}: [{},{}] | c={}".format(index, a, b, c))
if function(a) * function(c) > 0:
a = c
else:
b = c
time.sleep(delay) # Delays for 'delay' seconds. Make anything more cool at class
index += 1
print('Found solution after {} iterations.'.format(index))
return c

@staticmethod
def run_sample():
""" A sample of running the Bisection algorithm """
return Bisection.run(function=f, a=-5, b=5, precision=0.0001)

class RegulaFalsi:
""" Regula Falsi method for a given function """
@staticmethod
def run(function, a: float, b: float, precision: float, max_iter: int) -> float:
""" run the bisection method of a function and start conditions [a,b] :param function: the function to run :param a: the smaller location :param b: the upper location :param max_iter: how many iteration we allow in the process :param precision: how accurate we want the answer :return: where function = 0 inside [a,b] """
# first, make sure b > a if b < a:
temp = a
a = b
b = temp
xn = (a + b) / 2
m = (b - a) / (function(b) - function(a))
for n in range(max_iter):
f_xn = function(xn)
if abs(f_xn) < precision:
print('Found solution after {} iterations.'.format(n))
return xn
xn = xn - f_xn * m
print("#{}: Xn = {}".format(n, xn))
time.sleep(delay)
print('Exceeded maximum iterations. No solution found.')
return "error"
@staticmethod
def run_sample():
""" A sample of running the Bisection algorithm """
return RegulaFalsi.run(function=f, a=-5, b=5, precision=0.0001, max_iter=100)

class Secant:
""" Secant method for a given function """
@staticmethod
def run(function, a: float, b: float, precision: float) -> float:
""" run the bisection method of a function and start conditions [a,b] :param function: the function to run :param a: the smaller location :param b: the upper location :param precision: how accurate we want the answer :return: where function = 0 inside [a,b] """ # first, make sure b > a
if b < a:
temp = a
a = b
b = temp
# make sure the function decrease and then increase
if not (function(b) > 0 > function(a) or function(a) > 0 > function(b)): return "error"
# make the calculation itself
index = 1
x0 = a
x1 = b
while abs(x1-x0) > precision:
try:
x2 = (x0 * function(x1) - x1 * function(x0)) / (function(x1) - function(x0))
print("#{}: [{},{}] | c={}".format(index, x0, x1, x2))
time.sleep(delay)
x0 = x1
x1 = x2
index += 1
except Exception as error:
break
print('Found solution after {} iterations.'.format(index))
return x1

@staticmethod
def run_sample():
""" A sample of running the Bisection algorithm """
return Secant.run(function=f, a=-5, b=5, precision=0.0001)

class NewtonRaphson:
""" Secant method for a given function """
@staticmethod
def run(function, div_function, init_x: float, precision: float, max_iter: int) -> float:
""" run the bisection method of a function and start conditions [a,b] :param function: the function to run :param div_function: the function's diversion to run :param init_x: the seed point of the run :param max_iter: how many iteration we allow in the process :param precision: how accurate we want the answer :return: where function = 0 inside [a,b] """
xn = init_x
for n in range(max_iter):
f_xn = function(xn)
if abs(f_xn) < precision:
print('Found solution after {} iterations.'.format(n))
return xn
divf_xn = div_function(xn)
if divf_xn == 0:
print('Zero derivative. No solution found.')
return "error"
xn = xn - f_xn / divf_xn
print("#{}: Xn = {}".format(n, xn))
time.sleep(delay)
print('Exceeded maximum iterations. No solution found.')
return "error"

@staticmethod
def run_sample():
""" A sample of running the Bisection algorithm """
return NewtonRaphson.run(function=f, div_function=df, precision=0.0001, init_x=1, max_iter=100)
